

country_event = {
	id = 500040
	title = "African Culture Italianized!"
	desc = "We have finally managed to coerce a people from our holdings to adopt our customs!"

	trigger = {
		tag = ITA
		has_country_flag = assim_bedouin
		NOT = {accepted_culture = bedouin}
		NOT = {has_country_modifier = ita_assim_timer}
	}

	mean_time_to_happen = {
		days = 1
	}

	option = {
		name = "Italians need not be white!"
		add_accepted_culture = bedouin
	}
}
